96def assign_csv_data_to_uesgraph(uesgraph_input, base_folder, mappings, logger=None): 

97 """ 

98 Reads CSV-pandapipes results (Junctions & Pipes) 

99 and saves them as Pandas Series in UESGraph. 

100 

101 base_folder must have these folders: 

102 - res_junction/ 

103 - res_pipe/ 

104 

105 Files and the names in the graph: 

106 res_junction/p_bar.csv → pressure 

107 res_junction/t_k.csv → temperature 

108 res_pipe/mdot_from_kg_per_s.csv → mflow 

109 res_pipe/v_mean_m_per_s.csv → velocity 

110 

111 Parameters 

112 ---------- 

113 uesgraph_input : uesgraphs.uesgraph.UESGraph object 

114 UESGraph to assign the data to 

115 base_folder : str 

116 Folder where the CSV results are stored 

117 logger : logging.Logger, optional 

118 Logger instance for debugging 

119  

120 Returns 

121 ------- 

122 uesgraphs.uesgraph.UESGraph object 

123 UESGraph with assigned data 

124 """ 

125 

126 if logger is None: 

127 logger = set_up_terminal_logger(f"{__name__}.SystemModelHeating.assign_csv_data_to_uesgraph") 

128 

129 base_folder = Path(base_folder) 

130 

131 # Mappings depend on supply_type 

132 supply_type = uesgraph_input.graph.get("supply_type", "supply") 

133 

134 logger.info(supply_type) 

135 

136 junction_map = mappings[supply_type]["junction"] 

137 pipe_map = mappings[supply_type]["pipe"] 

138 

139 logger.info(junction_map) 

140 # CSV --> Variable Name Mapping 

141 csv_to_var = { 

142 "p_bar": "pressure", 

143 "t_k": "temperature", 

144 "mdot_from_kg_per_s": "mflow", 

145 "v_mean_m_per_s": "velocity", 

146 "p_from_bar": "pressure_from", 

147 "p_to_bar": "pressure_to" 

148 } 

149 

150 # --- Junction-Data ----------------------------------------------------- 

151 junction_folder = base_folder / "res_junction" 

152 for fname, var_name in [("p_bar.csv", "pressure"), ("t_k.csv", "temperature")]: 

153 fpath = junction_folder / fname 

154 if not fpath.exists(): 

155 continue 

156 

157 df = pd.read_csv(fpath, index_col=0, sep=';') 

158 if var_name == "pressure": 

159 df = df*100000 

160 

161 for col in df.columns: 

162 col_idx = int(col) 

163 if col_idx in junction_map: 

164 node_id = junction_map[col_idx] 

165 uesgraph_input.nodes[node_id][var_name] = df[col].tolist() # Pandas Series 

166 

167 p_in = None 

168 # --- Pipe-Data --------------------------------------------------------- 

169 pipe_folder = base_folder / "res_pipe" 

170 for fname, var_name in [("mdot_from_kg_per_s.csv", "m_flow"), 

171 ("p_from_bar.csv", "pressure_from"), 

172 ("p_to_bar.csv", "pressure_to")]: 

173 fpath = pipe_folder / fname 

174 if not fpath.exists(): 

175 continue 

176 

177 df = pd.read_csv(fpath, index_col=0, sep=';') 

178 if var_name == "pressure_from": 

179 df = df*100000 

180 p_in = df 

181 continue 

182 elif var_name == "pressure_to": 

183 df = df*100000 

184 if p_in is not None: 

185 df = p_in - df 

186 var_name = "dp" 

187 else: 

188 df = df 

189 

190 for col in df.columns: 

191 col_idx = int(col) 

192 if col_idx in pipe_map: 

193 edge = pipe_map[col_idx] 

194 uesgraph_input.edges[edge][var_name] = df[col].tolist() # Pandas Series 

195 

196 return uesgraph_input

200def create_model( 

201 name, 

202 save_at, 

203 graph, 

204 stop_time, 

205 timestep, 

206 t_ground_prescribed=None, 

207 logger=None 

208): 

209 """Generic model generation for setup defined through the parameters 

210 

211 Parameters 

212 ---------- 

213 name : str 

214 Name of the model (First character will be capitalized, cannot start with digit) 

215 save_at : str 

216 Directory where to store the generated model results 

217 graph : uesgraphs.uesgraph.UESGraph 

218 Network graph with all necessary data for model generation 

219 stop_time : int 

220 Stop time of the simulation in seconds 

221 timestep : int 

222 Timestep of the simulation in seconds 

223 t_ground_prescribed : list 

224 List of ground temperatures for every time step 

225 logger : logging.Logger, optional 

226 Logger instance for debugging 

227 

228 Returns 

229 ------- 

230 new_model : spp.SystemModelHeating 

231 Generated system model 

232 graph : uesgraphs.uesgraph.UESGraph 

233 Graph of the network with assigned simulation results 

234 """ 

235 # Set up logging 

236 if logger is None: 

237 logger = set_up_file_logger(f"{__name__}.create_model", level=logging.DEBUG) 

238 

239 # Entry logging with critical parameters 

240 logger.info(f"=== Starting create_model for '{name}' ===") 

241 logger.info(f"Target directory: {save_at}") 

242 logger.info(f"Network type: {graph.graph.get('network_type', 'unknown')}") 

243 logger.debug(f"Simulation setup - stop_time: {stop_time}, timestep: {timestep}") 

244 

245 assert not name[0].isdigit(), "Model name cannot start with a digit" 

246 

247 logger.info("Creating SystemModelHeating instance") 

248 new_model = spp.SystemModelHeating(stop_time = stop_time, timesteps= timestep, network_type=graph.graph["network_type"],logger=logger) 

249 

250 logger.info("Importing UESGraph") 

251 new_model.import_from_uesgraph(graph, logger=logger) 

252 

253 if t_ground_prescribed is not None: 

254 logger.debug(f"Setting prescribed ground temperatures (length: {len(t_ground_prescribed)})") 

255 new_model.graph["T_ground"] = t_ground_prescribed 

256 new_model.ground_temp_data = pd.DataFrame( 

257 {"1.0 m": new_model.graph["T_ground"]} 

258 ) 

259 

260 #new_model.run_test_simulation() 

261 

262 new_model.run_timeseries_spp(save_at, logger=logger) 

263 

264 mappings = { 

265 "supply": { 

266 "junction": new_model.junction_map_supply, 

267 "pipe": new_model.pipe_map_supply, 

268 }, 

269 "return": { 

270 "junction": new_model.junction_map_return, 

271 "pipe": new_model.pipe_map_return, 

272 }, 

273 } 

274 

275 with open(Path(save_at) / "mappings.pkl", "wb") as f: 

276 pickle.dump(mappings, f) 

277 

278 graph = assign_csv_data_to_uesgraph(graph, save_at, mappings, logger=logger) 

279 

280 logger.info(f"=== create_model completed successfully for '{name}' ===") 

281 return new_model, graph

289 log_level=logging.DEBUG 

290 ): 

291 """ 

292 Convert an Urban Energy System Graph (UESGraph) to pandapipes model and simulate it. 

293 

294 This function processes a UESGraph, applies simplification, adds demand data, 

295 and generates the pandapipes results based on simulation parameters. 

296 

297 Parameters: 

298 ----------- 

299 uesgraph : UESGraph object or str 

300 The UESGraph object or path to a JSON file containing the UESGraph data 

301 simplification_level : int 

302 Level of simplification to apply to the UESGraph (higher = more simplified)

304 Directory path where output files will be saved 

305 sim_setup_path : str or Path 

306 Path to the simulation setup configuration file

309 input_dhw : str or Path

312 Path to cooling demand data 

313 ground_temp_path : str or Path 

314 Path to ground temperature data file 

315 logger : logging.Logger, optional 

316 Logger instance. If None, creates a new file logger in temp directory 

317 log_level : int, optional 

318 Logging level (default is logging.DEBUG). Only used if logger is None 

319 

320 Returns: 

321 -------- 

322 None 

323 

324 Raises: 

325 ------- 

326 FileNotFoundError: If required files are missing 

327 ValueError: If uesgraph parameter is invalid 

328 Exception: For various processing errors with detailed messages 

329 """ 

330 # Set up logging configuration 

331 if logger is None: 

332 logger = set_up_file_logger("pandapipesSim", level=int(log_level)) 

333 

334 # Step 0: Resolve all paths to absolute paths (prevents issues with os.chdir in to_json) 

335 logger.debug("Resolving all input paths to absolute paths") 

336 workspace = os.path.abspath(str(workspace)) 

337 sim_setup_path = os.path.abspath(str(sim_setup_path)) 

338 input_heating = os.path.abspath(str(input_heating)) 

339 input_dhw = os.path.abspath(str(input_dhw))

341 ground_temp_path = os.path.abspath(str(ground_temp_path))

343 logger.debug(f"Resolved workspace: {workspace}") 

344 logger.debug(f"Resolved sim_setup_path: {sim_setup_path}") 

345 logger.debug(f"Resolved ground_temp_path: {ground_temp_path}") 

346 

347 # Step 1: Validate input files and folders 

348 logger.info("Validating files") 

349 paths_to_check = [sim_setup_path, input_heating, input_dhw, input_cooling, ground_temp_path] 

350 existing_files, missing_files = validate_paths(paths_to_check) 

351 if missing_files: 

352 logger.error(f"Missing files: {missing_files}") 

353 raise FileNotFoundError("Some files are missing. Please check the logs for details.") 

354 logger.debug(f"Existing files: {existing_files}") 

355 

356 # Step 2: Load simulation configuration from Excel 

357 logger.info("Loading simulation settings from Excel") 

358 sim_params = load_simulation_settings_from_excel(sim_setup_path, logger) 

359 logger.debug(f"Simulation settings loaded: {sim_params}") 

360 

361 # Step 3: Initialize or load the UESGraph from file or object 

362 logger.info("Initialize UESGraph") 

363 

364 if isinstance(uesgraph, (str, os.PathLike, Path)): 

365 if str(uesgraph).endswith(".json"): 

366 try: 

367 uesgraph_path = str(uesgraph) 

368 uesgraph = UESGraph()

370 logger.info(f"UESGraph loaded from JSON: {uesgraph_path}") 

371 except Exception as e: 

372 raise Exception(f"While loading UESGraph from JSON: {e}") 

373 else: 

374 raise ValueError(f"Value for uesgraph: {uesgraph} is nor uesgraph object nor valid JSON path") 

375 elif isinstance(uesgraph, UESGraph): 

376 # Normalize UESGraph via JSON 

377 logger.info("Normalize UESGraph via JSON") 

378 name = "uesgraphs_origin" 

379 origin_path = Path(workspace) / f"{name}.json" 

380 

381 try:

383 uesgraph.to_json( 

384 path=str(workspace),

391 # Reload from JSON to normalize internal structures 

392 normalized_graph = UESGraph() 

393 normalized_graph.from_json( 

394 path=str(origin_path), 

395 network_type="heating" 

396 ) 

397 

398 # Replace with normalized version 

399 uesgraph = normalized_graph 

400 logger.info("UESGraph normalized successfully") 

401 

402 except Exception as e: 

403 logger.error(f"Failed to normalize UESGraph: {e}") 

404 raise 

405 else: 

406 raise ValueError(f"uesgraph must be either a JSON path or UESGraph object, got: {type(uesgraph)}") 

407 

408 # Step 3.1: Write simulation parameters to graph (needed for connector time-series generation) 

409 logger.info("Writing simulation parameters to graph") 

410 uesgraph.graph['stop_time'] = float(sim_params['stop_time']) 

411 uesgraph.graph['timestep'] = sim_params.get('timestep', 3600) # Default 3600s = 1h 

412 uesgraph.graph["network_type"] = "heating" 

413 logger.debug(f"Set graph parameters: stop_time={uesgraph.graph['stop_time']}, timestep={uesgraph.graph['timestep']}")

432 if edges_without_names > 0: 

433 logger.info(f"Generated names for {edges_without_names}/{edges_total} edges that were missing names")

437 try:

587 """ 

588 Load simulation settings from Excel 'Simulation' sheet.

597 Returns 

598 ------- 

599 sim_params : dict 

600 Dictionary of simulation parameters 

601 

602 Raises 

603 ------ 

604 ValueError 

605 If required simulation parameters are missing 

606 """ 

607 if logger is None: 

608 logger = set_up_terminal_logger(f"{__name__}.load_simulation_settings_from_excel") 

609 

610 sim_params = _load_excel(excel_path, 'Simulation', logger) 

611 

612 # Validate required simulation parameters

656def load_component_parameters(excel_path, component_type): 

657 """ 

658 Load component parameters from an Excel file. 

659  

660 Reads a specific sheet from an Excel file and returns parameters as a dictionary. 

661 Expected Excel structure: 

662 - Column A: Parameter (parameter names) 

663 - Column B: Value (parameter values) 

664  

665 Parameters 

666 ---------- 

667 excel_path : str or Path 

668 Path to the Excel file containing component parameters 

669 component_type : str 

670 Type of component, must be one of: 'pipes', 'supply', 'demands', 'simulation' 

671 This determines which sheet to read from the Excel file 

672  

673 Returns 

674 ------- 

675 dict 

676 Dictionary with parameter names as keys and their values 

677 Returns empty dict if sheet not found 

678  

679 Raises 

680 ------ 

681 FileNotFoundError 

682 If the Excel file does not exist 

683 ValueError 

684 If the component_type is not valid or Excel structure is incorrect 

685  

686 Examples 

687 -------- 

688 >>> params = load_component_parameters('parameters.xlsx', 'pipes') 

689 >>> print(params['dp_nominal']) 

690 0.10 

691 """ 

692 # Validate component type 

693 valid_types = ['Pipes', 'Supply', 'Demands', 'Simulation'] 

694 if component_type not in valid_types: 

695 raise ValueError( 

696 f"Invalid component_type '{component_type}'. " 

697 f"Must be one of: {', '.join(valid_types)}" 

698 ) 

699 

700 # Check if file exists 

701 if not os.path.exists(excel_path): 

702 raise FileNotFoundError(f"Excel file not found: {excel_path}") 

703 

704 try: 

705 # Read the specific sheet 

706 df = pd.read_excel(excel_path, sheet_name=component_type) 

707 

708 # Validate Excel structure 

709 if len(df.columns) < 2: 

710 raise ValueError( 

711 f"Excel sheet '{component_type}' must have at least 2 columns " 

712 "(Parameter and Value)" 

713 ) 

714 

715 # Check if first row contains expected column names 

716 if 'Parameter' not in df.columns and 'parameter' not in df.columns.str.lower(): 

717 # Assume first two columns are Parameter and Value 

718 df.columns = ['Parameter', 'Value'] + list(df.columns[2:]) 

719 

720 # Create dictionary from Parameter and Value columns 

721 # Drop rows where Parameter is NaN 

722 df_clean = df[['Parameter', 'Value']].dropna(subset=['Parameter']) 

723 

724 # Convert to dictionary 

725 param_dict = dict(zip(df_clean['Parameter'], df_clean['Value'])) 

726 

727 # Convert and clean values 

728 import math 

729 for key, value in param_dict.items(): 

730 # Convert NaN to None 

731 if isinstance(value, float) and math.isnan(value): 

732 param_dict[key] = None 

733 # Try to convert string values to appropriate types 

734 elif isinstance(value, str): 

735 value_stripped = value.strip() 

736 

737 # Skip if it looks like a reference (@something) or template path/name 

738 if value_stripped.startswith('@') or '/' in value_stripped or '.' in value_stripped and not value_stripped.replace('.', '').replace('e', '').replace('E', '').replace('-', '').replace('+', '').isdigit(): 

739 continue 

740 

741 # Try boolean conversion 

742 if value_stripped.upper() in ('TRUE', 'FALSE'): 

743 param_dict[key] = value_stripped.upper() == 'TRUE' 

744 # Try numeric conversion (handles scientific notation like '1e5') 

745 else: 

746 try: 

747 # Try float first (handles both '123' and '1.23' and '1e5') 

748 param_dict[key] = float(value_stripped) 

749 except ValueError: 

750 # Keep as string if conversion fails (e.g., template names) 

751 pass 

752 

753 return param_dict 

754 

755 except ValueError as e: 

756 if "Worksheet named" in str(e): 

757 # Sheet doesn't exist 

758 logging.warning( 

759 f"Sheet '{component_type}' not found in {excel_path}. " 

760 f"Returning empty dictionary." 

761 ) 

762 return {} 

763 else: 

764 raise 

765 

766 except Exception as e: 

767 raise Exception( 

768 f"Error reading Excel file {excel_path}, sheet '{component_type}': {e}" 

769 )

824def assign_demand_data(uesgraph, input_paths_dict, input_types = ["heating", "cooling", "dhw"],demand_mode = 0): 

825 """ 

826 Assigns energy demand data to buildings in a UES (Urban Energy Systems) graph. 

827  

828 This function reads demand profiles from CSV files and assigns them to building nodes 

829 in the graph. For buildings without specific demand data, it uses a fallback profile 

830 (dummy demand) based on the first building in the respective CSV file. The function 

831 can handle heating, cooling, and domestic hot water (DHW) demands. 

832 

833 Parameters 

834 ---------- 

835 uesgraph : ueesgraphs Graph 

836 The urban energy system graph containing building nodes. 

837 input_paths_dict : dict 

838 Dictionary containing file paths for each demand type. 

839 Expected keys: 'heating', 'cooling', 'dhw'. 

840 input_types : list, optional 

841 List of demand types to process. Default is ["heating", "cooling", "dhw"]. 

842 demand_mode : int, optional 

843 Mode for demand calculation. Currently only mode 0 is implemented, 

844 which combines heating and DHW demands for peak load calculation. 

845 

846 Returns 

847 ------- 

848 tuple 

849 - Updated UES graph with demand data assigned to building nodes 

850 - Message string containing information about the data assignment process 

851  

852 Notes 

853 ----- 

854 Building nodes are updated with the following attributes: 

855 - input_heat: List of heating demand values 

856 - input_cool: List of cooling demand values 

857 - input_dhw: List of DHW demand values 

858 - max_demand_heating: Maximum combined heating/DHW demand 

859  

860 The function handles missing data by: 

861 1. Using a dummy profile if a building is not found in demand data 

862 2. Tracking which buildings exist in CSVs but not in the graph 

863 """ 

864 

865 msg = "Following Messages occured while loading demand data: \n" 

866 

867 #Define empty dictionaries for demand data 

868 dict_inputs = {} 

869 dummy_demand = {} 

870 

871 try: 

872 for input_type in input_types: 

873 df = pd.read_csv(input_paths_dict[input_type], 

874 parse_dates=True, 

875 index_col=0) 

876 dict_inputs[input_type] = df.rename(columns=lambda x: x.lower()) 

877 

878 #Store dummy demand (first data column) for missing buildings 

879 dummy_demand[input_type] = df[df.columns[1]].tolist() 

880 except FileNotFoundError as e: 

881 print(f"Failed to load demand data: {e}") 

882 

883 missing_bldgs = dict_inputs["heating"].columns.to_list() 

884 

885 #Process each building 

886 for bldg_node in uesgraph.nodelist_building: 

887 if uesgraph.nodes[bldg_node]["is_supply_heating"]: 

888 continue 

889 

890 bldg_name = uesgraph.nodes[bldg_node]["name"] 

891 demands = {} 

892 

893 #Get damand profiles for each type, ues dummy if missing 

894 for input_type in input_types: 

895 try: 

896 demands[input_type] = dict_inputs[input_type][bldg_name].tolist() 

897 if input_type == "heating": 

898 missing_bldgs.remove(bldg_name) 

899 except KeyError: 

900 demands[input_type] = dummy_demand[input_type] 

901 msg += f"Building {bldg_name} not found in {input_type} data. Using dummy demand." 

902 

903 #Save demands to node 

904 node = uesgraph.nodes[bldg_node] 

905 node.update({ 

906 "input_dhw": demands["dhw"], 

907 "input_heat": demands["heating"], 

908 "input_cool": demands["cooling"] 

909 }) 

910 if len(missing_bldgs)>0: 

911 msg += f"Missing buildings: {missing_bldgs}" 

912 else: 

913 msg += "No missing buildings found." 

914 

915 return uesgraph, msg

960def save_setup_params_to_csv(setup_params, sim_name, sim_model_dir): 

961 """ 

962 Save simulation setup parameters to a CSV file. 

963  

964 This function takes a single row from the simulation setup Excel file (as a dictionary) 

965 and saves it as a CSV file in a specified directory, with parameters as rows and their  

966 values as a single column. 

967  

968 Parameters: 

969 ----------- 

970 setup_params : dict 

971 Dictionary containing simulation setup parameters (a single row from the Excel setup) 

972 sim_name : str 

973 Name of the current simulation, used for filename generation 

974 sim_model_dir : str 

975 Directory where to save the CSV file 

976  

977 Returns: 

978 -------- 

979 str 

980 Path to the saved CSV file 

981 """ 

982 # Ensure the target directory exists 

983 os.makedirs(sim_model_dir, exist_ok=True) 

984 

985 # Define the path to the CSV file with correct naming 

986 csv_filename = f"{sim_name}.csv" 

987 csv_path = os.path.join(sim_model_dir, csv_filename) 

988 

989 # Convert the dictionary to a DataFrame and transpose it 

990 # This makes parameters appear as rows instead of columns 

991 params_df = pd.DataFrame([setup_params]).T 

992 

993 # Reset index to convert the parameter names to a column 

994 params_df.reset_index(inplace=True) 

995 

996 # Set column names 

997 params_df.columns = ["Parameter", "Value"] 

998 

999 # Check if any parameter value is excessively long 

1000 # This is optional but helps prevent unwieldy CSV files 

1001 for idx, row in params_df.iterrows(): 

1002 if isinstance(row["Value"], str) and len(row["Value"]) > 500: 

1003 # Truncate extremely long values with a note 

1004 params_df.at[idx, "Value"] = f"{row['Value'][:500]}... [truncated]" 

1005 

1006 # Save to CSV with semicolon separator 

1007 params_df.to_csv(csv_path, sep=";", index=False) 

1008 

1009 return csv_path
