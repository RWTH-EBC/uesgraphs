29def add_more_networks(example_district): 

30 """Adds an electric grid to the simple_dhc_model example 

31 

32 Parameters 

33 ---------- 

34 example_district : uesgraphs.uesgraph.UESGraph object 

35 An UESGraph containing 4 buildings connected to 2 heating supplies and 

36 2 cooling supplies via 2 heating and cooling networks each from 

37 `simple_dhc_model()` 

38 

39 Returns 

40 ------- 

41 example_district : uesgraphs.uesgraph.UESGraph object 

42 An UESGraph containing 8 buildings and various supplies and networks 

43 to illustrate UESGraph use 

44 """ 

45 

46 # Node numbers of the buildings are retrieved and saved. 

47 

48 building_1 = example_district.nodes_by_name["building_1"] 

49 building_2 = example_district.nodes_by_name["building_2"] 

50 building_3 = example_district.nodes_by_name["building_3"] 

51 building_4 = example_district.nodes_by_name["building_4"] 

52 

53 # Add electricity network 

54 # To generate a electricity supply building the 

55 # example_district.add_building function is used. Nine 

56 # parameters are available. The name and the position of the building node 

57 # need to be set. If the building should be a supply station this can be 

58 # set by one of the following parameters: is_supply_heating, 

59 # is_supply_cooling, is_supply_electricity, is_supply_gas and 

60 # is_supply_other. Below a electricity supply building node called 

61 # `supply_electricity` is added. 

62 

63 supply_electricity = example_district.add_building( 

64 position=Point(6, 2), is_supply_electricity=True 

65 ) 

66 

67 # To generate a network node the function example_district.add_network_node 

68 # is used. A network node should not be placed at positions where there 

69 # is already a node of the same network or a building node. Eight 

70 # parameters are available. network_type defines the network into which 

71 # to add the node.The position is compulsory. Below four network nodes for 

72 # a electricity network are added. 

73 

74 en_1 = example_district.add_network_node( 

75 network_type="electricity", position=Point(5, 2.5) 

76 ) 

77 en_2 = example_district.add_network_node( 

78 network_type="electricity", position=Point(4, 2.5) 

79 ) 

80 en_3 = example_district.add_network_node( 

81 network_type="electricity", position=Point(3, 2.5) 

82 ) 

83 en_4 = example_district.add_network_node( 

84 network_type="electricity", position=Point(2, 2.5) 

85 ) 

86 

87 # The nodes of the electricity network are connected using the 

88 # example_district.add_edge function. 

89 

90 example_district.add_edge(en_1, building_4) 

91 example_district.add_edge(en_2, building_3) 

92 example_district.add_edge(en_3, building_2) 

93 example_district.add_edge(en_4, building_1) 

94 example_district.add_edge(en_1, en_2) 

95 example_district.add_edge(en_2, en_3) 

96 example_district.add_edge(en_3, en_4) 

97 example_district.add_edge(en_1, supply_electricity) 

98 

99 # Add more buildings supplied by gas and electricity 

100 

101 building_5 = example_district.add_building(position=Point(7, 3)) 

102 building_6 = example_district.add_building(position=Point(7, 4)) 

103 building_7 = example_district.add_building(position=Point(7, 5)) 

104 

105 # Add three network nodes of the type electricity. 

106 

107 en_5 = example_district.add_network_node( 

108 network_type="electricity", position=Point(6.5, 3) 

109 ) 

110 en_6 = example_district.add_network_node( 

111 network_type="electricity", position=Point(6.5, 4) 

112 ) 

113 en_7 = example_district.add_network_node( 

114 network_type="electricity", position=Point(6.5, 5) 

115 ) 

116 

117 # Using example_district.add_edge the nodes of electricity network are 

118 # connected. 

119 

120 example_district.add_edge(en_5, en_6) 

121 example_district.add_edge(en_6, en_7) 

122 example_district.add_edge(en_5, supply_electricity) 

123 example_district.add_edge(en_5, building_5) 

124 example_district.add_edge(en_6, building_6) 

125 example_district.add_edge(en_7, building_7) 

126 

127 # Add a network of type 'gas'. 

128 # Add a new supply building for gas called `supply_gas`. 

129 

130 supply_gas = example_district.add_building( 

131 position=Point(8, 1), is_supply_gas=True 

132 ) 

133 

134 # Add network nodes for gas network. 

135 

136 gn_1 = example_district.add_network_node( 

137 network_type="gas", position=Point(8, 3) 

138 ) 

139 gn_2 = example_district.add_network_node( 

140 network_type="gas", position=Point(8, 4) 

141 ) 

142 gn_3 = example_district.add_network_node( 

143 network_type="gas", position=Point(8, 5) 

144 ) 

145 

146 # Connect nodes of gas network. 

147 

148 example_district.add_edge(gn_1, gn_2) 

149 example_district.add_edge(gn_2, gn_3) 

150 example_district.add_edge(gn_1, supply_gas) 

151 example_district.add_edge(gn_1, building_5) 

152 example_district.add_edge(gn_2, building_6) 

153 example_district.add_edge(gn_3, building_7) 

154 

155 # Add a network of type 'other' 

156 # Generate a supply building of type 'other'. 

157 

158 supply_other = example_district.add_building( 

159 position=Point(7.5, 7), is_supply_other=True 

160 ) 

161 

162 # Generate two network nodes of type 'other' using the add_network_node 

163 # function. 

164 

165 on_1 = example_district.add_network_node( 

166 network_type="others", position=Point(7.5, 5) 

167 ) 

168 on_2 = example_district.add_network_node( 

169 network_type="others", position=Point(7.5, 4) 

170 ) 

171 

172 # Connect the nodes of network 'other'. 

173 

174 example_district.add_edge(on_1, on_2) 

175 example_district.add_edge(on_1, supply_other) 

176 example_district.add_edge(on_1, building_7) 

177 example_district.add_edge(on_2, building_6) 

178 

179 # Add street nodes using the example_district.add_street_node function. 

180 # In this function five parameters are available. The position of the 

181 # street can be set. Also the resolution can be given which means the 

182 # minimum distance between two points. The check_overlap parameter is set 

183 # true by default. Below four streets are generated in block 1 and four 

184 # street are generated in block 2. 

185 

186 sn_11 = example_district.add_street_node(position=Point(1, 2.5), block=1) 

187 sn_12 = example_district.add_street_node(position=Point(6, 2.5), block=1) 

188 sn_13 = example_district.add_street_node(position=Point(6, 3.5), block=1) 

189 sn_14 = example_district.add_street_node(position=Point(1, 3.5), block=1) 

190 

191 sn_21 = example_district.add_street_node(position=Point(6, 5.5), block=2) 

192 sn_22 = example_district.add_street_node(position=Point(6, 2.5), block=2) 

193 sn_23 = example_district.add_street_node(position=Point(8, 2.5), block=2) 

194 sn_24 = example_district.add_street_node(position=Point(8, 5.5), block=2) 

195 

196 # Connecting streets of block 1 

197 

198 example_district.add_edge(sn_11, sn_12) 

199 example_district.add_edge(sn_12, sn_13) 

200 example_district.add_edge(sn_13, sn_14) 

201 example_district.add_edge(sn_14, sn_11) 

202 

203 # Connecting street of block 2 

204 

205 example_district.add_edge(sn_21, sn_22) 

206 example_district.add_edge(sn_22, sn_23) 

207 example_district.add_edge(sn_23, sn_24) 

208 example_district.add_edge(sn_24, sn_21) 

209 

210 return example_district
