29def check_input_file(file_path: str, logger=None) -> str: 

30 """ 

31 Check and prepare input file for processing. 

32  

33 Handles different file formats and converts .mat files to .gzip if needed. 

34  

35 Args: 

36 file_path: Path to the input file 

37  

38 Returns: 

39 Path to the processed file (usually .gzip format) 

40  

41 Raises: 

42 ValueError: If file doesn't exist or conversion fails 

43 """ 

44 if logger is None: 

45 logger = set_up_terminal_logger(f"{__name__}.check_input_file") 

46 

47 if not file_path: 

48 raise ValueError("File path cannot be empty") 

49 

50 base_path = os.path.splitext(file_path)[0] 

51 gzip_path = f"{base_path}.gzip" 

52 

53 # Check for gzip first (preferred format) 

54 if os.path.exists(gzip_path): 

55 logger.info(f"Found existing gzip file: {gzip_path}") 

56 return gzip_path 

57 

58 # Check for .mat file and convert 

59 mat_path = f"{base_path}.mat" 

60 if os.path.exists(mat_path): 

61 try: 

62 logger.info(f"Converting .mat file to parquet: {mat_path}") 

63 gzip_new = mat_to_parquet(save_as=base_path, fname=mat_path, with_unit=False) 

64 logger.info(f"Successfully converted .mat file to: {gzip_new}") 

65 return gzip_new 

66 except Exception as e: 

67 logger.error(f"Failed to convert .mat file: {mat_path}") 

68 raise ValueError(f"Could not convert .mat file to parquet: {mat_path}") from e 

69 

70 # Finally check if original file exists 

71 if not os.path.exists(file_path): 

72 raise ValueError(f"File does not exist: {file_path}") 

73 

74 logger.info(f"Using original file: {file_path}") 

75 return file_path

77def validate_columns_exist(file_path: str, required_columns: List[str], 

78 logger: Optional[logging.Logger] = None) -> Set[str]: 

79 """ 

80 Check if all required columns exist in the simulation data file. 

81  

82 Args: 

83 file_path: Path to the parquet/simulation file 

84 required_columns: List of exact column names that must exist 

85 logger: Logger instance (optional) 

86  

87 Returns: 

88 Set of available columns from the file 

89  

90 Raises: 

91 KeyError: If any required columns are missing 

92 ValueError: If file cannot be read 

93 """ 

94 if logger is None: 

95 logger = set_up_terminal_logger(f"{__name__}.validate_columns_exist") 

96 

97 logger.info(f"Validating {len(required_columns)} required columns in: {file_path}") 

98 

99 # Read file metadata (no data loading) 

100 try: 

101 parquet_file = pq.ParquetFile(file_path) 

102 available_columns = set(parquet_file.schema.names) 

103 logger.debug(f"File contains {len(available_columns)} total columns") 

104 except Exception as e: 

105 error_msg = f"Could not read file metadata: {e}" 

106 logger.error(error_msg) 

107 raise ValueError(error_msg) 

108 

109 # Check for missing columns 

110 required_set = set(required_columns) 

111 missing_columns = required_set - available_columns 

112 

113 if missing_columns: 

114 missing_list = sorted(missing_columns) 

115 error_msg = f"Missing required columns: {missing_list}" 

116 logger.error(error_msg) 

117 

118 # Raise KeyError with first missing column for auto-retry compatibility 

119 first_missing = missing_list[0] 

120 raise KeyError(first_missing) 

121 

122 logger.info("SUCCESS: All required columns found in data file") 

123 return available_columns

125def process_simulation_result(file_path: str, filter_list: List[str], 

126 chunk_size: int = 100000, logger=None) -> Generator[pd.DataFrame, None, None]: 

127 """ 

128 Process a parquet file in chunks to reduce memory usage. 

129  

130 Args: 

131 file_path: Path to the parquet file 

132 filter_list: List of column patterns to filter 

133 chunk_size: Number of rows to process at once 

134 logger: Optional logger instance 

135  

136 Yields: 

137 pd.DataFrame: Processed chunks of the parquet file 

138 """ 

139 if logger is None: 

140 logger = set_up_terminal_logger(f"{__name__}.process_parquet_file") 

141 

142 # Step 1: Check if file exists and convert .mat if needed 

143 processed_file_path = check_input_file(file_path=file_path, logger=logger) 

144 

145 # Step 1: Validate all columns exist (will raise KeyError if missing) 

146 available_columns = validate_columns_exist(processed_file_path, required_columns=filter_list, logger=logger) 

147 

148 # Step 2: Check if any columns match the filter_list 

149 logger.info(f"Starting parquet file processing: {processed_file_path}") 

150 logger.debug(f"Filter patterns: {filter_list}") 

151 logger.debug(f"Chunk size: {chunk_size}") 

152 

153 try: 

154 # Read parquet file metadata to get columns 

155 parquet_file = pq.ParquetFile(processed_file_path, 

156 thrift_string_size_limit=2_000_000_000, 

157 thrift_container_size_limit=2_000_000_000) 

158 chunks = [] 

159 total_rows = 0 

160 

161 

162 # Read and process the file in chunks 

163 for batch in parquet_file.iter_batches(batch_size=chunk_size, columns=filter_list): 

164 total_rows += 1 

165 chunk_df = batch.to_pandas() 

166 chunks.append(chunk_df) 

167 if len(chunks) % 10 == 0: # Log every 10 chunks 

168 logger.debug(f"Loaded {len(chunks)} chunks, {total_rows} rows so far") 

169 

170 if not chunks: 

171 logger.warning("No data loaded from file") 

172 return pd.DataFrame() 

173 

174 result_df = pd.concat(chunks, axis = 0, ignore_index=True) 

175 

176 logger.info(f"Successfully loaded {len(result_df)} rows, {len(result_df.columns)} columns") 

177 logger.debug(f"DataFrame memory usage: {result_df.memory_usage(deep=True).sum() / 1024**2:.2f} MB") 

178 return result_df 

179 except Exception as e: 

180 logger.error(f"Error processing parquet file {processed_file_path}: {str(e)}") 

181 raise e

187def prepare_DataFrame(df, base_date=datetime(2024, 1, 1), time_interval="15min", 

188 start_date=None, end_date=None, logger=None) -> pd.DataFrame: 

189 """ 

190 Prepare a DataFrame with a datetime index using customizable parameters. 

191  

192 Parameters: 

193 ----------- 

194 df : pandas.DataFrame 

195 The DataFrame to be processed 

196 base_date : datetime, optional 

197 The starting date for the index (default: 2024-01-01) 

198 time_interval : str, optional 

199 Frequency of the time intervals (e.g., '15min', '1h', '30min', default: '15min') 

200 start_date : datetime, optional 

201 If provided, slice the DataFrame from this date (inclusive) 

202 end_date : datetime, optional 

203 If provided, slice the DataFrame until this date (inclusive) 

204 logger : logging.Logger, optional 

205 Logger instance for logging operations 

206  

207 Returns: 

208 -------- 

209 DataFrame: A DataFrame containing the data from the parquet file for the specified time period. 

210 """ 

211 if logger is None: 

212 logger = set_up_terminal_logger(f"{__name__}.prepare_DataFrame") 

213 

214 logger.info(f"Preparing DataFrame with {len(df)} rows and {len(df.columns)} columns") 

215 logger.debug(f"Parameters - base_date: {base_date}, time_interval: {time_interval}") 

216 logger.debug(f"Date filtering - start_date: {start_date}, end_date: {end_date}") 

217 

218 try: 

219 # Create datetime index with specified frequency  

220 logger.debug(f"Creating datetime index with frequency '{time_interval}'") 

221 datetime_index = pd.date_range(start=base_date, periods=len(df), freq=time_interval) 

222 logger.debug(f"Created datetime index from {datetime_index[0]} to {datetime_index[-1]}") 

223 

224 # Set the index of the DataFrame to the datetime index 

225 df.index = datetime_index 

226 df.index.name = 'DateTime' 

227 logger.info(f"Applied datetime index to DataFrame") 

228 

229 # Filter by date range if specified 

230 original_length = len(df) 

231 if start_date is not None and end_date is not None: 

232 logger.info(f"Filtering DataFrame from {start_date} to {end_date}") 

233 df = df.loc[start_date:end_date] 

234 elif start_date is not None: 

235 logger.info(f"Filtering DataFrame from {start_date} onwards") 

236 df = df.loc[start_date:] 

237 elif end_date is not None: 

238 logger.info(f"Filtering DataFrame up to {end_date}") 

239 df = df.loc[:end_date] 

240 

241 # Log filtering results if any filtering was applied 

242 if len(df) != original_length: 

243 logger.info(f"Date filtering applied: {original_length} -> {len(df)} rows ({len(df)/original_length*100:.1f}% retained)") 

244 if len(df) == 0: 

245 logger.warning("Date filtering resulted in empty DataFrame - check date range parameters") 

246 else: 

247 logger.debug("No date filtering applied") 

248 

249 logger.info(f"Successfully prepared DataFrame: {len(df)} rows, index range: {df.index[0]} to {df.index[-1]}") 

250 return df 

251 

252 except ValueError as e: 

253 error_msg = f"Error creating date range with frequency {time_interval} and base date {base_date}. Original error: {e}" 

254 logger.error(error_msg) 

255 raise ValueError(error_msg) from e 

256 except Exception as e: 

257 error_msg = f"Unexpected error in prepare_DataFrame: {str(e)}" 

258 logger.error(error_msg) 

259 raise

301def build_filter_list_pipe(graph, mask, logger=None): 

302 """ 

303 Build a list of filter variables for pipes in a district heating network graph. 

304  

305 This function extracts patterns from a mask data structure and formats them 

306 with specific pipe codes and type information for each edge (pipe) in the graph. 

307 This is useful for filtering and analyzing district heating networks. 

308  

309 Args: 

310 graph : uesgraphs object. 

311 mask (dict): Hierarchical data structure containing filter patterns. 

312 Expected categories are 'edge' and 'node': 

313 - 'edge': Dict with direct pattern values 

314 - 'node': Dict with nested pattern values 

315 logger (logging.Logger, optional): Logger instance for debug output. 

316 Will be created automatically if None. 

317  

318 Returns: 

319 list: List of formatted variable names for all pipes in the graph, 

320 based on the mask patterns. 

321  

322 Example: 

323 >>> mask = { 

324 ... 'edge': {'pressure': 'p_{pipe_code}_{type}'}, 

325 ... 'node': {'inlet': {'temp': 'T_in_{pipe_code}_{type}'}} 

326 ... } 

327 >>> filter_list = build_filter_list_pipe(graph, mask) 

328 >>> print(filter_list) 

329 ['p_PIPE001_supply', 'T_in_PIPE001_supply', ...] 

330  

331 Raises: 

332 KeyError: If pipe edges don't have a 'name' attribute 

333 AttributeError: If graph doesn't have edges attribute 

334 """ 

335 # Initialize logger if not provided 

336 if logger is None: 

337 logger = set_up_terminal_logger("BuildFilterListPipe") 

338 

339 # Collection of all simulation patterns from the mask structure 

340 simulation_patterns = [] 

341 

342 # Iterate through all categories in the mask 

343 for category_name, category_data in mask.items(): 

344 if category_name not in ["edge", "node"]: 

345 logger.warning(f"Unknown category '{category_name}' in mask, skipping") 

346 continue 

347 

348 if category_name == "edge": 

349 # Edge category: direct extraction of pattern values 

350 # Example: {'m_flow': 'p_{pipe_code}_{type}'} -> ['p_{pipe_code}_{type}'] 

351 simulation_patterns.extend(category_data.values()) 

352 logger.debug(f"Added {len(category_data)} edge patterns") 

353 

354 elif category_name == "node": 

355 # Node category: nested structure - extract all port patterns 

356 # Example: {'temperature': {'port_a': 'T_in_{pipe_code}_{type}'}}  

357 # -> ['T_in_{pipe_code}_{type}'] 

358 for port_name, attribute_patterns in category_data.items(): 

359 if isinstance(attribute_patterns, dict): 

360 simulation_patterns.extend(attribute_patterns.values()) 

361 logger.debug(f"Added {len(attribute_patterns)} node patterns " 

362 f"for port '{port_name}'") 

363 else: 

364 logger.warning(f"Expected dict for node port '{port_name}', " 

365 f"got {type(attribute_patterns)}") 

366 

367 logger.info(f"Extracted {len(simulation_patterns)} simulation patterns total") 

368 

369 # Get type prefix from graph (e.g., 'supply', 'return') 

370 type_prefix = get_supply_type_prefix(graph) 

371 logger.debug(f"Using type prefix: '{type_prefix}'") 

372 

373 # List for all generated filter variables 

374 filter_variables = [] 

375 

376 # Generate filter variables for each edge (pipe) in the graph 

377 for edge in graph.edges: 

378 try: 

379 # Extract pipe code from edge attributes 

380 pipe_code = graph.edges[edge]["name"] 

381 

382 # Generate a variable for each simulation pattern 

383 for pattern in simulation_patterns: 

384 # Format pattern with specific values 

385 variable_name = pattern.format( 

386 pipe_code=pipe_code, 

387 type=type_prefix 

388 ) 

389 filter_variables.append(variable_name) 

390 

391 logger.debug(f"Generated {len(simulation_patterns)} variables " 

392 f"for pipe '{pipe_code}'") 

393 

394 except KeyError as e: 

395 logger.error(f"Edge {edge} missing required attribute: {e}") 

396 raise 

397 except Exception as e: 

398 logger.error(f"Error processing edge {edge}: {e}") 

399 raise 

400 

401 logger.info(f"Created filter list with {len(filter_variables)} entries " 

402 f"for {len(graph.edges)} pipes") 

403 

404 return filter_variables

422 2: ['pipe001.port_b', 'pipe003.port_a']} 

423 Source: [Method name if known, otherwise leave empty] 

424 mask: Mask dictionary containing node configuration for intensive properties 

425 time_index: Time step index to extract from df (default: 0) 

426 logger: Logger instance (optional, creates terminal logger if None) 

427 """ 

428 if logger is None: 

429 logger = set_up_terminal_logger(f"{__name__}.assign_node_values") 

430 

431 # Extract node configuration 

432 node_config = mask.get("node", {})

434 logger.error("No 'node' configuration found in mask") 

435 return

461def _assign_attributes_to_node(graph, node_id, node_ports, df, config, 

462 type_suffix, stats, logger): 

463 """Assign all attributes to a single node. 

464 config: {"attribute": {"port_suffix": "pattern_with_{pipe_code}"}} 

465 ex.: {"temperature": {"port_a": "networkModel.pipe{pipe_code}{type}.sta_a.T"}} 

466 """ 

467 

468 for attribute_name, port_patterns in config.items(): 

469 """Collect values for a specific attribute from all relevant ports. 

470 attribute_name: e.g. "temperature" 

471 port_patterns: e.g. {"port_a": "networkModel.pipe{pipe_code}{type}.sta_a.T", 

472 "port_b": "networkModel.pipe{pipe_code}{type}.sta_b.T"} 

473 """ 

474 series_list = [] 

475 for port in node_ports: 

476 

477 pipe_name, port_suffix = _parse_port_identifier(port) 

478 if port_suffix in port_patterns: 

479 pattern = port_patterns[port_suffix] 

480 column_name = pattern.format(pipe_code=pipe_name, type=type_suffix) 

481 

482 if column_name in df.columns: 

483 series = df[column_name] 

484 series_list.append(series) 

485 else: 

486 logger.debug(f"Column not found: {column_name}") 

487 

488 if len(series_list) == 0: 

489 logger.debug(f"No data found for {attribute_name} at node {node_id}") 

490 continue 

491 

492 # Check if all series are identical 

493 if len(series_list) > 1: 

494 all_equal = all(series_list[0].equals(series) for series in series_list[1:]) 

495 if not all_equal: 

496 logger.warning(f"Node {node_id}: Inconsistent {attribute_name} time series found") 

497 stats['inconsistency_count'] += 1 

498 

499 # Use first series as result 

500 graph.nodes[node_id][attribute_name] = series_list[0]

502def _parse_port_identifier(port): 

503 """Parse port identifier to extract pipe name and port suffix.""" 

504 port_parts = port.split(".") 

505 if len(port_parts) < 2: 

506 raise ValueError(f"Invalid port format: {port}. Expected 'pipe_name.port_suffix'") 

507 return port_parts[0].replace("pipe", ""), port_parts[1]

521def validate_edge_attributes(graph, edge_attributes, reference_df, logger=None): 

522 """ 

523 Validates graph edge attributes against a reference DataFrame. 

524  

525 Checks if the length of attribute arrays in edges matches the  

526 number of rows in the reference DataFrame. 

527  

528 Args: 

529 graph: uesgraphs 

530 edge_attributes (dict): Dictionary containing edge attributes to validate 

531 reference_df (pd.DataFrame): Reference DataFrame for length comparison 

532 logger (logging.Logger, optional): Logger instance. If None, a  

533 terminal logger will be created. 

534  

535 Returns: 

536 bool: True if all validations pass, False otherwise 

537  

538 Raises: 

539 ValueError: For critical validation errors 

540 """ 

541 if logger is None: 

542 logger = set_up_terminal_logger(f"{__name__}.validate_edge_attributes") 

543 

544 logger.info("Starting edge attribute validation...") 

545 

546 expected_length = reference_df.shape[0] 

547 validation_passed = True 

548 errors = [] 

549 

550 # Validate edge attributes 

551 logger.info(f"Validating edge attributes for {len(graph.edges)} edges...") 

552 

553 if edge_attributes: 

554 for edge_idx, edge in enumerate(graph.edges): 

555 for edge_attr in edge_attributes.keys(): 

556 if edge_attr in graph.edges[edge]: 

557 actual_length = len(graph.edges[edge][edge_attr]) 

558 

559 if actual_length != expected_length: 

560 error_msg = (

562 f"length {actual_length} != expected length {expected_length}" 

563 ) 

564 logger.error(error_msg) 

565 errors.append(error_msg) 

566 validation_passed = False 

567 else: 

568 logger.debug( 

569 f"Edge {edge} - Attribute '{edge_attr}': OK " 

570 f"(length: {actual_length})" 

571 ) 

572 else: 

573 warning_msg = f"Edge {edge} - Attribute '{edge_attr}' not found" 

574 logger.warning(warning_msg) 

575 else: 

576 logger.warning("No edge attributes provided for validation") 

577 

578 # Validation summary 

579 if validation_passed: 

580 logger.info("SUCCESS: Edge attribute validation completed successfully") 

581 else: 

582 logger.error(f"FAILED: Edge attribute validation failed: {len(errors)} errors") 

583 for error in errors[:5]: # Show maximum 5 errors in summary 

584 logger.error(f" - {error}") 

585 if len(errors) > 5: 

586 logger.error(f" ... and {len(errors) - 5} more errors") 

587 

588 return validation_passed

590def validate_node_attributes(graph, node_attributes, reference_df, logger=None): 

591 """

624 for node_attr in node_attributes.keys(): 

625 if node_attr in graph.nodes[node]: 

626 actual_length = len(graph.nodes[node][node_attr]) 

627 

628 if actual_length != expected_length: 

629 error_msg = ( 

630 f"Node {node} - Attribute '{node_attr}': " 

631 f"length {actual_length} != expected length {expected_length}" 

632 ) 

633 logger.error(error_msg) 

634 errors.append(error_msg) 

635 validation_passed = False 

636 else: 

637 logger.debug( 

638 f"Node {node} - Attribute '{node_attr}': OK " 

639 f"(length: {actual_length})" 

640 ) 

641 else: 

642 warning_msg = f"Node {node} - Attribute '{node_attr}' not found" 

643 logger.warning(warning_msg) 

644 else: 

645 logger.warning("No node attributes provided for validation") 

646 

647 # Validation summary 

648 if validation_passed: 

649 logger.info("SUCCESS: Node attribute validation completed successfully") 

650 else: 

651 logger.error(f"FAILED: Node attribute validation failed: {len(errors)} errors") 

652 for error in errors[:5]: # Show maximum 5 errors in summary 

653 logger.error(f" - {error}") 

654 if len(errors) > 5: 

655 logger.error(f" ... and {len(errors) - 5} more errors") 

656 

657 return validation_passed

660def assess_dp_quality(graph, 

661 negligible_abs_threshold=1.0, 

662 negligible_rel_threshold=0.001, 

663 acceptable_abs_threshold=10.0, 

664 acceptable_rel_threshold=0.01): 

665 """ 

666 Assesses the quality of node assignments based on pressure and pressure difference 

667  

668 Parameters: 

669 ----------- 

670 graph : uesgraph 

671 The district heating network graph object 

672 negligible_abs_threshold : float, default=1.0 

673 Absolute threshold for negligible deviations (Pa) 

674 negligible_rel_threshold : float, default=0.001 

675 Relative threshold for negligible deviations (0.1%) 

676 acceptable_abs_threshold : float, default=10.0 

677 Absolute threshold for acceptable deviations (Pa) 

678 acceptable_rel_threshold : float, default=0.01 

679 Relative threshold for acceptable deviations (1%) 

680  

681 Returns: 

682 -------- 

683 dict 

684 Dictionary with categories 'negligible', 'acceptable', 'investigate' 

685 and corresponding edges with timestamp information 

686 """ 

687 stats = { 

688 'negligible': [], 

689 'acceptable': [], 

690 'investigate': [] 

691 } 

692 

693 for edge in graph.edges: 

694 node1, node2 = list(edge) 

695 p1 = graph.nodes[node1]["pressure"] 

696 p2 = graph.nodes[node2]["pressure"] 

697 dp_calc = p1 - p2 

698 dp_sim = graph.edges[edge]["dp"] 

699 

700 # Check for each timestamp 

701 for i in range(len(p1)): 

702 timestamp = p1.index[i] if hasattr(p1, 'index') else i 

703 

704 abs_diff = abs(dp_calc.iloc[i] - dp_sim.iloc[i]) 

705 rel_error = (abs_diff / abs(dp_sim.iloc[i]) 

706 if dp_sim.iloc[i] != 0 else float('inf')) 

707 

708 edge_info = { 

709 'edge': edge, 

710 'timestamp': timestamp, 

711 'abs_diff': abs_diff, 

712 'rel_error': rel_error, 

713 'dp_calculated': dp_calc.iloc[i], 

714 'dp_simulated': dp_sim.iloc[i] 

715 } 

716 

717 # Categorization based on thresholds 

718 if (abs_diff < negligible_abs_threshold or 

719 rel_error < negligible_rel_threshold): 

720 stats['negligible'].append(edge_info) 

721 

722 elif (abs_diff < acceptable_abs_threshold and 

723 rel_error < acceptable_rel_threshold): 

724 stats['acceptable'].append(edge_info) 

725 

726 else: 

727 stats['investigate'].append(edge_info) 

728 

729 return stats

735def assign_data_pipeline( 

736 graph: ug.UESGraph, 

737 simulation_data_path: Union[str, Path], 

738 start_date: datetime, 

739 end_date: datetime, 

740 time_interval: str, 

741 MASK: Optional[Dict[str, str]] = None, 

742 aixlib_version: str = "2.1.0", 

743 system_model_path: Optional[Union[str, Path]] = None, 

744 node_to_port_mapping: Optional[Dict] = None, 

745 logger: Optional[logging.Logger] = None 

746) -> ug.UESGraph: 

747 """ 

748 Assign simulation data to a UESGraph network. 

749  

750 This function processes simulation results and assigns time series data 

751 to network components (nodes and edges). It supports two modes: 

752  

753 1. **Full assignment** (with node data): Requires either `node_to_port_mapping` 

754 or `system_model_path` to map simulation variables to graph nodes 

755 2. **Edge-only assignment**: When no mapping is available, only assigns 

756 data to edges (mass flows, pressure drops) 

757  

758 Args: 

759 graph: UESGraph instance to assign data to 

760 simulation_data_path: Path to simulation results (.mat or .parquet) 

761 start_date: Start date for data processing 

762 end_date: End date for data processing  

763 time_interval: Time interval for resampling (e.g., "15min", "1H") 

764 No default - user must specify explicitly 

765 MASK: Custom variable name masks. If None, uses AixLib standard masks 

766 aixlib_version: AixLib version for standard masks (default: "2.1.0") 

767 system_model_path: Path to system model JSON (for creating port mapping) 

768 node_to_port_mapping: Pre-computed mapping from nodes to simulation ports 

769 logger: Logger instance. If None, creates a new file logger 

770  

771 Returns: 

772 UESGraph instance with assigned simulation data 

773  

774 Raises: 

775 FileNotFoundError: If simulation_data_path doesn't exist 

776 ValueError: If graph has no name set or data validation fails 

777 KeyError: If required simulation variables are missing 

778  

779 Notes: 

780 - Either `node_to_port_mapping` OR `system_model_path` is required  

781 for full data assignment including nodes 

782 - If both are None, only edge data (mass flows) will be assigned 

783 - Graph must have a name set in graph.graph["name"] 

784  

785 Example: 

786 >>> import uesgraphs as ug 

787 >>> from datetime import datetime 

788 >>>  

789 >>> # Load your network 

790 >>> graph = ug.UESGraph()  

791 >>> graph.from_json("network.json", network_type="heating") 

792 >>> graph.graph["name"] = "my_network" 

793 >>>  

794 >>> # Assign simulation data 

795 >>> graph_with_data = assign_data_pipeline( 

796 ... graph=graph, 

797 ... simulation_data_path="results.mat", 

798 ... start_date=datetime(2024, 1, 1), 

799 ... end_date=datetime(2024, 1, 7),  

800 ... time_interval="15min", 

801 ... system_model_path="system_model.json" 

802 ... ) 

803 >>>  

804 >>> # With custom masks 

805 >>> custom_masks = { 

806 ... "m_flow": "custom.pipe{pipe_code}.flow", 

807 ... "p_a": "custom.pipe{pipe_code}.pressure_in",  

808 ... "p_b": "custom.pipe{pipe_code}.pressure_out" 

809 ... } 

810 >>> graph_with_data = assign_data_pipeline( 

811 ... graph=graph, 

812 ... simulation_data_path="results.mat",  

813 ... start_date=datetime(2024, 1, 1), 

814 ... end_date=datetime(2024, 1, 7), 

815 ... time_interval="1H", 

816 ... MASK=custom_masks 

817 ... ) 

818 """ 

819 # Set up logging 

820 if logger is None: 

821 logger = set_up_file_logger("assign_data_pipeline", level=logging.INFO) 

822 

823 # Convert simulation_data_path to Path object  

824 simulation_data_path = Path(simulation_data_path) 

825 

826 # Validate graph has a name 

827 network_name = graph.graph.get("name") 

828 if not network_name: 

829 raise ValueError("Graph must have a name set in graph.graph['name']") 

830 

831 # Check supply type from graph 

832 supply_type = graph.graph.get("supply_type", "supply") 

833 supply_type_prefix = {"supply": "", "return": "R"} 

834 

835 logger.info("="*70) 

836 logger.info("STARTING DATA ASSIGNMENT PIPELINE") 

837 logger.info("="*70) 

838 logger.info(f"Network: {network_name}") 

839 logger.info(f"Graph: {len(graph.nodes)} nodes, {len(graph.edges)} edges") 

840 logger.info(f"Simulation data: {simulation_data_path}") 

841 logger.info(f"Time period: {start_date} to {end_date}") 

842 logger.info(f"Time interval: {time_interval}") 

843 logger.info(f"Supply type: {supply_type}") 

844 

845 # Determine assignment mode based on available mapping options 

846 assignment_mode = _determine_assignment_mode( 

847 system_model_path, node_to_port_mapping, logger 

848 ) 

849 

850 try: 

851 # Step 1: Determine variable masks 

852 logger.info("Step 1/6: Setting up variable masks") 

853 if MASK is None: 

854 MASK = AIXLIB_MASKS[aixlib_version] 

855 logger.info(f"SUCCESS: Using AixLib {aixlib_version} standard masks") 

856 else: 

857 logger.info("SUCCESS: Using custom variable masks") 

858 

859 # Step 2: Create or use port mapping (if available) 

860 port_mapping = None 

861 if assignment_mode == "full": 

862 logger.info("Step 2/6: Setting up port mapping for node assignment") 

863 

864 if node_to_port_mapping is not None: 

865 port_mapping = node_to_port_mapping 

866 logger.info("SUCCESS: Using provided node-to-port mapping") 

867 

868 elif system_model_path is not None: 

869 system_model_path = Path(system_model_path) 

870 if not system_model_path.exists(): 

871 raise FileNotFoundError(f"System model file not found: {system_model_path}") 

872 

873 sysm_graph = ut.load_system_model_from_json(str(system_model_path)) 

874 port_mapping = graph_transformation.map_system_model_to_uesgraph(sysm_graph, graph) 

875 logger.info(f"SUCCESS: Created port mapping from system model ({len(port_mapping)} components)") 

876 

877 elif assignment_mode == "edges_only": 

878 logger.info("Step 2/6: Skipping port mapping - edge-only assignment mode") 

879 logger.warning("WARNING: No node data will be assigned (temperatures, pressures)") 

880 

881 # Step 3: Process simulation data 

882 logger.info("Step 3/6: Processing simulation data") 

883 

884 # Check if simulation data file exists and convert .mat files if needed 

885 if not simulation_data_path.exists(): 

886 raise FileNotFoundError(f"Simulation data file not found: {simulation_data_path}") 

887 

888 # Convert .mat to .gzip if needed and get the processed file path 

889 processed_simulation_path = check_input_file(file_path=str(simulation_data_path), logger=logger) 

890 logger.info(f"SUCCESS: Using processed simulation file: {processed_simulation_path}") 

891 

892 # Build filter list for required variables 

893 filter_list = build_filter_list_pipe(graph, mask=MASK, logger=logger) 

894 logger.info(f"SUCCESS: Built filter list with {len(filter_list)} variables") 

895 

896 # Validate that all required columns exist in the processed file 

897 column_validation = validate_columns_exist( 

898 file_path=processed_simulation_path, 

899 required_columns=filter_list, 

900 logger=logger 

901 ) 

902 if not column_validation: 

903 raise KeyError("Required simulation variables not found in data file") 

904 

905 # Load and process simulation results 

906 df = process_simulation_result( 

907 file_path=processed_simulation_path, 

908 filter_list=filter_list, 

909 logger=logger 

910 ) 

911 logger.info(f"SUCCESS: Loaded simulation data: {df.shape[0]} timesteps") 

912 

913 # Step 4: Prepare DataFrame 

914 logger.info("Step 4/6: Preparing time series data") 

915 df = prepare_DataFrame( 

916 df, 

917 start_date=start_date, 

918 end_date=end_date, 

919 time_interval=time_interval, 

920 logger=logger 

921 ) 

922 logger.info(f"SUCCESS: Prepared DataFrame: {df.shape[0]} timesteps after filtering") 

923 

924 # Step 5: Assign data to graph components 

925 logger.info("Step 5/6: Assigning data to graph components") 

926 

927 if assignment_mode == "full": 

928 # Assign values to nodes (temperature and pressure) 

929 assign_node_values(graph, df, port_mapping, MASK, logger=logger) 

930 logger.info(f"SUCCESS: Assigned node data (temperature, pressure) to {len(graph.nodes)} nodes") 

931 

932 # Assign values to edges (mass flow, pressure drop) - always done 

933 assign_edge_data(graph, MASK, df) 

934 logger.info(f"SUCCESS: Assigned edge data (mass flow, pressure drop) to {len(graph.edges)} edges") 

935 

936 # Step 6: Validate results 

937 logger.info("Step 6/6: Validating assignment results") 

938 

939 validate_edge_attributes(graph, MASK["edge"], df, logger=logger) 

940 logger.info("SUCCESS: Edge validation completed successfully") 

941 

942 if assignment_mode == "full": 

943 validate_node_attributes(graph,MASK["node"],df, logger=logger) 

944 

945 ## Additional test to asses node assignment based on pressures 

946 stats = assess_dp_quality(graph) 

947 if len(stats['investigate']) > 1: 

948 logger.warning("WARNING: Critical pressure difference deviations found!") 

949 else: 

950 logger.info("SUCCESS: Full validation completed successfully") 

951 

952 

953 logger.info("="*70) 

954 logger.info("PIPELINE COMPLETED SUCCESSFULLY") 

955 logger.info(f"SUCCESS: Network '{network_name}' ready for analysis") 

956 logger.info(f"SUCCESS: Data period: {df.index.min()} to {df.index.max()}") 

957 logger.info(f"SUCCESS: Time resolution: {time_interval}") 

958 logger.info(f"SUCCESS: Assignment mode: {assignment_mode}") 

959 logger.info("="*70) 

960 

961 return graph 

962 

963 except Exception as e: 

964 logger.error(f"Pipeline failed: {str(e)}") 

965 logger.error("="*70) 

966 raise

969def _determine_assignment_mode( 

970 system_model_path: Optional[Union[str, Path]], 

971 node_to_port_mapping: Optional[Dict], 

972 logger: logging.Logger 

973) -> str: 

974 """ 

975 Determine the assignment mode based on available mapping options. 

976  

977 Returns: 

978 "full": Full assignment including nodes (requires mapping) 

979 "edges_only": Only edge assignment (no node data) 

980 """ 

981 if node_to_port_mapping is not None: 

982 logger.info("Port mapping provided → Full assignment mode") 

983 return "full" 

984 elif system_model_path is not None: 

985 logger.info("System model provided → Full assignment mode") 

986 return "full" 

987 else: 

988 logger.warning("No port mapping or system model → Edge-only assignment mode") 

989 logger.warning("Node temperatures and pressures will NOT be assigned") 

990 return "edges_only"
