4 """

6 uesgraphs 

7 Args:

9 The system model graph where pipes are nodes and

11 node_id (str): The ID or name of the node for which to find connected pipes 

12  

13 """

15 for pipe in graph_sysm.nodelist_pipe: 

16 if graph_sysm.nodes[pipe].get("node_0") == node_id or graph_sysm.nodes[pipe].get("node_1") == node_id:

20def get_ports_for_network_node(graph_sysm, pipe_nodes_sysm): 

21 """ 

22 Identifies all unique connection ports for a set of pipes that form a node in the UES graph. 

23 

24 This function addresses the mapping between two different graph representations: 

25 1. System model graph (graph_sysm): Pipes are modeled as nodes, and connections between 

26 pipes are modeled as edges with port information. 

27 2. UES graph: Pipes are modeled as edges, and connections between pipes are modeled as nodes. 

28 

29 In the system model, state changes occur within pipes (nodes), while connections maintain 

30 constant states. These connections correspond to nodes in the UES graph. This function 

31 finds all connections between the provided pipes, which represent a single node in the  

32 UES graph. 

33 

34 Args: 

35 graph_sysm (uesgraphs.systemmodels.systemmodelheating.SystemModelHeating):  

36 The system model graph where pipes are nodes and  

37 connections are edges 

38 pipe_nodes_sysm (list): List of pipe IDs that are connected to the same node in  

39 the UES graph 

40 

41 Returns: 

42 set: All unique port identifiers where these pipes connect, representing points  

43 with identical states in the physical model 

44 """ 

45 # Collect all unique ports connected to this node (set avoids duplicates) 

46 ports_per_node = set() 

47 

48 # Check all possible pipe combinations (unordered pairs) 

49 for edge1, edge2 in itertools.combinations(pipe_nodes_sysm, 2): 

50 # If these pipes are connected in the system model 

51 if graph_sysm.has_edge(edge1, edge2): 

52 # Add the connection ports to our collection 

53 # These ports represent points with identical state in the model 

54 ports_per_node.add(graph_sysm.edges[edge1, edge2]['con1']) 

55 ports_per_node.add(graph_sysm.edges[edge1, edge2]['con2']) 

56 

57 return ports_per_node

59def get_ports_for_bldg_node(graph_sysm,name): 

60 """ 

61 Identifies the connection ports for a building node (consumer or supplier). 

62  

63 Building nodes are special terminal nodes that have only one pipe connection. 

64 Their port handling differs from junction nodes. 

65 

66 Args: 

67 graph_sysm (uesgraphs.systemmodels.systemmodelheating.SystemModelHeating):  

68 The system model graph where pipes are nodes and  

69 connections are edges 

70 name: str: The name of the building node for which to find connected ports 

71 Returns: 

72 set: A set of ports connected to the building node 

73 """ 

74 for bldg_model in graph_sysm.nodelist_building: 

75 ports_per_bldg = set() 

76 # Find the building node in system model graph that matches the uesgraph node 

77 if name == graph_sysm.nodes[bldg_model]["name"]: 

78 # Get the edges connected to the building node 

79 adj_edges = graph_sysm.edges(bldg_model) 

80 if len(adj_edges) > 1: 

81 print(f"Interesting case, cause multiple pipes are connected to building {name}") 

82 # Since edges in the system models graph correspond to nodes in uesgraph both connections  

83 # should get the same simulation results 

84 for edge in adj_edges: 

85 ports_per_bldg.add(graph_sysm.edges[edge]['con1']) 

86 ports_per_bldg.add(graph_sysm.edges[edge]['con2']) 

87 return ports_per_bldg

89def validate_structural_node_port_mapping(uesgraph, node_to_ports_mapping): 

90 """ 

91 Validates the structural integrity of the node-to-ports mapping. 

92  

93 This function performs a structural validation only, checking that: 

94 1. All nodes in the UES graph have a corresponding entry in the mapping 

95 2. Every node has at least two ports assigned (the minimum required for any connection) 

96  

97 Note: This is only a structural validation and does not verify the actual values 

98 or physical correctness of the ports. A complete validation would require comparing 

99 actual port values against expected physical constraints. 

100  

101 Args: 

102 uesgraph (uesraphs object): The UES graph containing all nodes to validate 

103 node_to_ports_mapping (dict): The mapping from node IDs to port lists 

104  

105 Raises: 

106 ValueError: If any structural validation criteria are not met, with specific 

107 details about which nodes failed validation and why 

108 """ 

109 # Check 1: All nodes from the UES graph must be present in the mapping 

110 missing_nodes = set(uesgraph.nodes) - set(node_to_ports_mapping.keys()) 

111 if missing_nodes: 

112 raise ValueError(f"Missing port mappings for nodes: {missing_nodes}") 

113 

114 # Check 2: All nodes should have at least two ports (minimum requirement for connections) 

115 invalid_nodes = [] 

116 for node, ports in node_to_ports_mapping.items(): 

117 # Every connection needs at least two ports 

118 if len(ports) < 2: 

119 invalid_nodes.append(f"Node {node} has only {len(ports)} ports, minimum required is 2") 

120 

121 if invalid_nodes: 

122 raise ValueError("Structural validation failed:\n" + "\n".join(invalid_nodes))

124def map_system_model_to_uesgraph(graph_sysm,uesgraph): 

125 """ 

126 Maps each node in the UES graph to its corresponding ports in the system model graph. 

127  

128 This function handles the transformation between two different graph representations: 

129 1. UES graph: Models the physical network where pipes are edges and junctions are nodes 

130 2. System model graph: Models the simulation components where pipes are nodes and 

131 connections between pipes are edges  

132 

133 

134 Args: 

135 graph_sysm (uesgraphs.systemmodels.systemmodelheating.SystemModelHeating):  

136 The system model graph where pipes are nodes and  

137 connections are edges 

138 uesgraph (uesgraphs.uesgraph.UESGraph): The UES graph where pipes are edges and  

139 connections are nodes 

140  

141 Returns: 

142 dict: Mapping from UES graph node IDs to lists of corresponding system model ports 

143  

144 Raises: 

145 ValueError: If a non-building node has only one pipe connection (open end) 

146 """ 

147 node_to_ports_mapping = {} 

148 

149 # Iterate over all nodes in the UES graph 

150 for node in uesgraph.nodes: 

151 name = uesgraph.nodes[node]["name"] 

152 # Get all pipes connected to this node in the system model 

153 relevant_pipes = get_pipes_for_node(graph_sysm, name) 

154 #print(f"For node {node} found {relevant_pipes} pipes") 

155 

156 # Handle based on the number of connected pipes 

157 if len(relevant_pipes) == 1: 

158 # Terminal nodes should only be buildings (consumers or suppliers) 

159 if uesgraph.nodes[node]["node_type"] == "building": 

160 node_to_ports_mapping[node] = list(get_ports_for_bldg_node(graph_sysm, name)) 

161 else: 

162 raise ValueError(f"Node {node} has an open end, meaning just one pipe is connected to it, while its not a consumer or supplier") 

163 else: 

164 # For junction nodes, find all unique ports connected to the pipes 

165 node_to_ports_mapping[node]= list(get_ports_for_network_node(graph_sysm, relevant_pipes)) 

166 

167 #Validate the mapping structure 

168 validate_structural_node_port_mapping(uesgraph, node_to_ports_mapping) 

169 

170 return node_to_ports_mapping
